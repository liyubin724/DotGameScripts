local LogUtil = CS.Dot.Log.LogUtil
local table = table

LoggerMessageType = {
    Info = 0,
    Warning = 1,
    Error = 2,
}

----private----
Logger = {}
Logger.__index = Logger

----public----

function Logger.SetEnable(isErrorEnable,isWarningEnable,isInfoEnable)
    Logger.isErrorEnable = isErrorEnable or true
    Logger.isWarningEnable = isWarningEnable or true
    Logger.isInfoEnable = isInfoEnable or true
end

function Logger.Error(loggerName,data)
    if Logger.isErrorEnable then
        PrintLogMessage(loggerName,LoggerMessageType.Error,"LUA:"..table.tostring(data))
    end
end

function Logger.Warning(loggerName,data)
    if Logger.isWarningEnable then
        PrintLogMessage(loggerName,LoggerMessageType.Warning,"LUA:"..table.tostring(data))
    end
end

function Logger.Info(loggerName,data)
    if Logger.isInfoEnable then
        PrintLogMessage(loggerName,LoggerMessageType.Info,"LUA:"..table.tostring(data))
    end
end
